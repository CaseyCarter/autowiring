151  template<class T>
152  static SlotInformationStackLocation PushStackLocation(T* pSpace) {
153  auto* pStump = &SlotInformationStump<T>::s_stump;
154 
155  // If we're already initialized, then we have nothing to do. This line is an optimization; if there
156  // is a race here, then the worst-case scenario is an unneeded sequence of memory allocations that
157  // will only ever be referenced by this stack location.
158  if(pStump->bInitialized)
159  return SlotInformationStackLocation(pStump);
160 
161  // New stack location to enclose this stump. This stack location may be concurrent with respect
162  // to other threads, but only one thread will succeed in colonizing this stump with a chain of
163  // slot entries
164  return SlotInformationStackLocation(pStump, pSpace, sizeof(T));
165  }
