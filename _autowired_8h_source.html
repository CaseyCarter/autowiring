141  template<class Fn>
142  void NotifyWhenAutowired(Fn fn) {
143  // Trivial initial check:
144  if(*this) {
145  fn();
146  return;
147  }
148 
149  // We pass a null shared_ptr, because we do not want this slot to attempt any kind of unregistration when
150  // it goes out of scope. Instead, we will manage its entire registration lifecycle, and
151  // retain full ownership over the object until we need to destroy it.
152  auto newHead = new AutowirableSlotFn<T, Fn>(std::shared_ptr<CoreContext>(), std::forward<Fn>(fn));
153 
154  // Append to our list in a lock-free way. This is a fairly standard way to do a lock-free append to
155  // a singly linked list; the only unusual aspect is the use of "this" as a tombstone indicator (IE,
156  // to signify that the list should no longer be used).
157  AutowirableSlot<T>* pFirstChild;
158  do {
159  // Obtain the current first child, and keep it here so we know what to exchange out with later:
160  pFirstChild = m_pFirstChild;
161 
162  // Determine if we've been tombstoned at this piont:
163  if(pFirstChild == this) {
164  // Trivially satisfy, and then return. This might look like a leak, but it's not, because we know
165  // that Finalize is going to destroy the object.
166  newHead->SatisfyAutowiring(*this);
167  newHead->Finalize();
168  return;
169  }
170 
171  // Try to set the forward link to the current head, and then update our own flink;
172  newHead->SetFlink(pFirstChild);
173  } while(!m_pFirstChild.compare_exchange_weak(pFirstChild, newHead, std::memory_order_acquire));
174  }
175 
176  // Base overrides:
177  DeferrableAutowiring* ReleaseDependentChain(void) override {
178  // Rip the head off the list, replace it with a tombstone:
179  auto pFirstChild = m_pFirstChild.exchange(this, std::memory_order_relaxed);
180 
181  // If we got the tombstone back, we have nothing to return. Otherwise return the element.
182  return pFirstChild == this ? nullptr : pFirstChild;
183  }
