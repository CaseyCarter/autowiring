208  template<class Fn>
209  void NotifyWhenAutowired(Fn fn) {
210  // Trivial initial check:
211  if(*this) {
212  fn();
213  return;
214  }
215 
216  // We pass a null shared_ptr, because we do not want this slot to attempt any kind of unregistration when
217  // it goes out of scope. Instead, we will manage its entire registration lifecycle, and
218  // retain full ownership over the object until we need to destroy it.
219  auto newHead = new AutowirableSlotFn<T, Fn>(std::shared_ptr<CoreContext>(), std::forward<Fn>(fn));
220 
221  // Append to our list in a lock-free way. This is a fairly standard way to do a lock-free append to
222  // a singly linked list; the only unusual aspect is the use of "this" as a tombstone indicator (IE,
223  // to signify that the list should no longer be used).
224  AutowirableSlot<T>* pFirstChild;
225  do {
226  // Obtain the current first child, and keep it here so we know what to exchange out with later:
227  pFirstChild = m_pFirstChild;
228 
229  // Determine if we've been tombstoned at this piont:
230  if(pFirstChild == this) {
231  // Trivially satisfy, and then return. This might look like a leak, but it's not, because we know
232  // that Finalize is going to destroy the object.
233  newHead->SatisfyAutowiring(*this);
234  newHead->Finalize();
235  return;
236  }
237 
238  // Try to set the forward link to the current head, and then update our own flink;
239  newHead->SetFlink(pFirstChild);
240  } while(!m_pFirstChild.compare_exchange_weak(pFirstChild, newHead, std::memory_order_acquire));
241  }
242 
243  // Base overrides:
244  DeferrableAutowiring* ReleaseDependentChain(void) override {
245  // Rip the head off the list, replace it with a tombstone:
246  auto pFirstChild = m_pFirstChild.exchange(this, std::memory_order_relaxed);
247 
248  // If we got the tombstone back, we have nothing to return. Otherwise return the element.
249  return pFirstChild == this ? nullptr : pFirstChild;
250  }
