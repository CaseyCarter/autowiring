101  bool FireCurried(const Fn& fn, Args&... args) const {
102  std::unique_lock<std::mutex> lk(m_lock);
103  int deleteCount = m_numberOfDeletions;
104 
105  // Collection of contexts that need to be torn down in the event of an exception. Using a list,
106  // here, instead of a vector because this list is expected to almost always be empty, and we can
107  // avoid having to include <vector> by using a list instead.
108  std::list<std::weak_ptr<CoreContext>> teardown;
109 
110  for (auto it = m_st.begin(); it != m_st.end();){
111  JunctionBoxEntry<T> currentEvent(*it);
112 
113  lk.unlock();
114  try {
115  fn(*currentEvent.m_ptr, args...);
116  }
117  catch (...) {
118  teardown.push_back(ContextDumbToWeak(currentEvent.m_owner));
119 
120  // If T doesn't inherit Object, then we need to cast to a unifying type which does
121  typedef typename SelectTypeUnifier<T>::type TActual;
122  this->FilterFiringException(autowiring::fast_pointer_cast<TActual>(currentEvent.m_ptr));
123  }
124  lk.lock();
125 
126  // Increment iterator correctly even if it's been invalidated
127  if (deleteCount == m_numberOfDeletions){
128  ++it;
129  }
130  else {
131  it = m_st.upper_bound(currentEvent);
132  deleteCount = m_numberOfDeletions;
133  }
134  }
135  if (teardown.empty())
136  // Nobody threw any exceptions, end here
137  return true;
138
