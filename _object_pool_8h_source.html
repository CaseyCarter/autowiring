118  std::shared_ptr<T> Wrap(T* pObj) {
119  // Fill the shared pointer with the object we created, and ensure that we override
120  // the destructor so that the object is returned to the pool when it falls out of
121  // scope.
122  size_t poolVersion = m_poolVersion;
123  auto monitor = m_monitor;
124  std::function<void(T&)> final = m_final;
125 
126  auto retVal = std::shared_ptr<T>(
127  pObj,
128  [poolVersion, monitor, final](T* ptr) {
129  // Finalize object before destruction or return to pool
130  final(*ptr);
131 
132  // Default behavior will be to destroy the pointer
133  std::unique_ptr<T> unique(ptr);
134 
135  // Hold the lock next, in order to ensure that destruction happens after the monitor
136  // lock is released.
137  std::lock_guard<std::mutex> lk(*monitor);
138 
139  if(monitor->IsAbandoned())
140  // Nothing we can do, monitor object abandoned already, just destroy the object
141  return;
142 
143  // Pass control to the Return method
144  static_cast<ObjectPool<T>*>(monitor->GetOwner())->Return(poolVersion, unique);
145  }
146  );
147 
148  // Initialize the issued object, now that a shared pointer has been created for it
149  m_initial(*pObj);
150 
151  // All done
152  return retVal;
153  }

155  void Return(size_t poolVersion, std::unique_ptr<T>& unique) {
156  // ASSERT: Object has already been finalized
157  // Always decrement the count when an object is no longer outstanding
158  assert(m_outstanding);
159  m_outstanding--;
160 
161  if(
162  // Pool versions have to match, or the object should be dumped
163  poolVersion == m_poolVersion &&
164 
165  // Object pool needs to be capable of accepting another object as an input
166  m_objs.size() < m_maxPooled
167  ) {
168  // Return the object to the pool:
169  m_objs.emplace_back(std::move(unique));
170  }
171 
172  // If the new outstanding count is less than or equal to the limit, wake up any waiters:
173  if(m_outstanding <= m_limit)
174  m_setCondition.notify_all();
175  }

184  std::shared_ptr<T> ObtainElementUnsafe(std::unique_lock<std::mutex>& lk) {
185  // Unconditionally increment the outstanding count:
186  m_outstanding++;
187 
188  // Cached, or construct?
189  if(m_objs.empty()) {
190  // Lock release, so construction does not have to be synchronized:
191  lk.unlock();
192 
193  // We failed to recover an object, create a new one:
194  auto obj = Wrap(m_alloc());
195  return obj;
196  }
197

199  std::shared_ptr<T> iObj = Wrap(m_objs.back().release()); // Takes ownership
200  m_objs.pop_back(); // Removes non-referencing object
201  return iObj;
202  }

336  void Preallocate(size_t reservation) {
337  if(reservation > m_maxPooled)
338  reservation = m_maxPooled;
339 
340  // Check out objects into our vector, allowing them to all return to the pool at once.
341  // This will populate the pool with the desired number of entities, and will trigger
342  // the desired failure condition if we wind up checking out more objects than are
343  // allowed for this pool.

345  while(reservation--)
