100 struct crh<construction_strategy::standard, T, Args...> {
101  // If T doesn't inherit Object, then we need to compose a unifying type which does
102  typedef typename SelectTypeUnifier<T>::type TActual;
103 
104  static_assert(!std::is_abstract<T>::value, "Cannot create a type which is abstract");
105  static_assert(!has_static_new<T, Args...>::value, "Can't inject member with arguments if it has a static New");
106 
107  static TActual* New(const CoreContext&, Args&&... args) {
108  // Allocate slot first before registration
109  auto* pSpace = Allocate<TActual>(nullptr);
110 
111  try {
112  // Push a new stack location so that all constructors from here know the injected type under construction
113  SlotInformationStackLocation loc(
114  &SlotInformationStump<T>::s_stump,
115  pSpace,
116  sizeof(TActual)
117  );
118 
119  // And now we create our space
120  return ::new (pSpace) TActual(std::forward<Args>(args)...);
121  }
122  catch(...) {
123  // Use our free method, which will correctly invoke the deallocator for this space
124  Free<TActual>(pSpace, nullptr);
125  throw;
126  }
127  }
