216  template<class T>
217  bool Get(const T*& out, int tshift=0) const {
218  DecorationKey key(auto_id<T>::key(), false, tshift);
219  const DecorationDisposition* pDisposition = GetDisposition(key);
220  if (pDisposition) {
221  switch (pDisposition->m_decorations.size()) {
222  case 0:
223  // No shared pointer decorations available, we have to try something else
224  break;
225  case 1:
226  // Single decoration, we can do what the user is asking
227  out = static_cast<const T*>(pDisposition->m_decorations[0]->ptr());
228  return true;
229  default:
230  ThrowMultiplyDecoratedException(key);
231  break;
232  }
233 
234  // Second-chance satisfaction with an immediate
235  if (pDisposition->m_pImmediate) {
236  out = (T*) pDisposition->m_pImmediate;
237  return true;
238  }
239  }
240 
241  out = nullptr;
242  return false;
243  }

433  template<class T, class... Ts>
434  void DecorateImmediate(const T& immed, const Ts&... immeds) {
435  // None of the inputs may be shared pointers--if any of the inputs are shared pointers, they must be attached
436  // to this packet via Decorate, or else dereferenced and used that way.
437  static_assert(
438  !is_any<is_shared_ptr<T>::value, is_shared_ptr<Ts>::value...>::value,
439  "DecorateImmediate must not be used to attach a shared pointer, use Decorate on such a decoration instead"
440  );
441 
442  // Perform standard decoration with a short initialization:
443  std::unique_lock<std::mutex> lk(m_lock);
444  DecorationDisposition* pTypeSubs[1 + sizeof...(Ts)] = {
445  &DecorateImmediateUnsafe(DecorationKey(auto_id<T>::key(), false, 0), &immed),
446  &DecorateImmediateUnsafe(DecorationKey(auto_id<Ts>::key(), false, 0), &immeds)...
447  };
448  lk.unlock();
449 
450 
451  // Pulse satisfaction:
452  MakeAtExit([this, &pTypeSubs] {
453  // Mark entries as unsatisfiable:
454  // IMPORTANT: isCheckedOut = true prevents subsequent decorations of this type
455  for(DecorationDisposition* pEntry : pTypeSubs) {
456  pEntry->m_pImmediate = nullptr;
457  pEntry->m_state = DispositionState::Unsatisfiable;
458  }
459 
460  // Now trigger a rescan to hit any deferred, unsatisfiable entries:
461  for (const std::type_info* ti : {&auto_id<T>::key(), &auto_id<Ts>::key()...}) {
462  MarkUnsatisfiable(DecorationKey(*ti, true, 0));
463  MarkUnsatisfiable(DecorationKey(*ti, false, 0));
464  }
465  }),
466  PulseSatisfaction(pTypeSubs, 1 + sizeof...(Ts));
467  }
