245  template<class T>
246  bool Get(const T*& out, int tshift=0) const {
247  autowiring::DecorationKey key(auto_id_t<T>{}, tshift);
248  const autowiring::DecorationDisposition* pDisposition = GetDisposition(key);
249  if (pDisposition) {
250  switch (pDisposition->m_decorations.size()) {
251  case 0:
252  // No shared pointer decorations available, we have to try something else
253  break;
254  case 1:
255  // Single decoration, we can do what the user is asking
256  out = static_cast<const T*>(pDisposition->m_decorations[0].ptr());
257  return true;
258  default:
259  ThrowMultiplyDecoratedException(key);
260  break;
261  }
262 
263  // Second-chance satisfaction with an immediate
264  if (pDisposition->m_pImmediate) {
265  out = (T*) pDisposition->m_pImmediate;
266  return true;
267  }
268  }
269 
270  out = nullptr;
271  return false;
272  }

553  void DecorateImmediate(const T& immed, const Ts&... immeds) {
554  // None of the inputs may be shared pointers--if any of the inputs are shared pointers, they must be attached
555  // to this packet via Decorate, or else dereferenced and used that way.
556  static_assert(
557  !autowiring::is_any<autowiring::is_shared_ptr<T>::value, autowiring::is_shared_ptr<Ts>::value...>::value,
558  "DecorateImmediate must not be used to attach a shared pointer, use Decorate on such a decoration instead"
559  );
560 
561  // Perform standard decoration with a short initialization:
562  std::unique_lock<std::mutex> lk(m_lock);
563  autowiring::DecorationDisposition* pTypeSubs[1 + sizeof...(Ts)] = {
564  &DecorateImmediateUnsafe(autowiring::DecorationKey(auto_id_t<T>(), 0), &immed),
565  &DecorateImmediateUnsafe(autowiring::DecorationKey(auto_id_t<Ts>(), 0), &immeds)...
566  };

568  // Pulse satisfaction:
569  MakeAtExit([this, &pTypeSubs] {
570  // Mark entries as unsatisfiable:
571  // IMPORTANT: isCheckedOut = true prevents subsequent decorations of this type
572  for(autowiring::DecorationDisposition* pEntry : pTypeSubs) {
573  pEntry->m_pImmediate = nullptr;
574  pEntry->m_state = autowiring::DispositionState::Complete;
575  }
576 
577  // Now trigger a rescan to hit any deferred, unsatisfiable entries:
578  autowiring::noop(
579  (MarkUnsatisfiable(autowiring::DecorationKey(auto_id_t<T>(), 0)), false),

582  }),
