406  template<class T, class... Ts>
407  void DecorateImmediate(const T& immed, const Ts&... immeds) {
408  // None of the inputs may be shared pointers--if any of the inputs are shared pointers, they must be attached
409  // to this packet via Decorate, or else dereferenced and used that way.
410  static_assert(
411  !is_any<is_shared_ptr<T>::value, is_shared_ptr<Ts>::value...>::value,
412  "DecorateImmediate must not be used to attach a shared pointer, use Decorate on such a decoration instead"
413  );
414 
415  // Perform standard decoration with a short initialization:
416  std::unique_lock<std::mutex> lk(m_lock);
417  DecorationDisposition* pTypeSubs[1 + sizeof...(Ts)] = {
418  &DecorateImmediateUnsafe(DecorationKey(auto_id<T>::key(), false, 0), &immed),
419  &DecorateImmediateUnsafe(DecorationKey(auto_id<Ts>::key(), false, 0), &immeds)...
420  };
421  lk.unlock();
422 
423 
424  // Pulse satisfaction:
425  MakeAtExit([this, &pTypeSubs] {
426  // Mark entries as unsatisfiable:
427  // IMPORTANT: isCheckedOut = true prevents subsequent decorations of this type
428  for(DecorationDisposition* pEntry : pTypeSubs) {
429  pEntry->m_pImmediate = nullptr;
430  pEntry->m_state = DispositionState::Unsatisfiable;
431  }
432 
433  // Now trigger a rescan to hit any deferred, unsatisfiable entries:
434 #if autowiring_USE_LIBCXX
435  for (const std::type_info* ti : {&auto_id<T>::key(), &auto_id<Ts>::key()...}) {
436  MarkUnsatisfiable(DecorationKey(*ti, true, 0));
437  MarkUnsatisfiable(DecorationKey(*ti, false, 0));
438  }
439 #else
440  bool dummy[] = {
441  (MarkUnsatisfiable(DecorationKey(auto_id<T>::key(), false, 0)), false),
442  (MarkUnsatisfiable(DecorationKey(auto_id<Ts>::key(), false, 0)), false)...
443  };
444  (void)dummy;
445 #endif
446  }),
447  PulseSatisfaction(pTypeSubs, 1 + sizeof...(Ts));
448  }
