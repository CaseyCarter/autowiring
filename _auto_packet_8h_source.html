403  template<class T, class... Ts>
404  void DecorateImmediate(const T& immed, const Ts&... immeds) {
405  // None of the inputs may be shared pointers--if any of the inputs are shared pointers, they must be attached
406  // to this packet via Decorate, or else dereferenced and used that way.
407  static_assert(
408  !is_any<is_shared_ptr<T>::value, is_shared_ptr<Ts>::value...>::value,
409  "DecorateImmediate must not be used to attach a shared pointer, use Decorate on such a decoration instead"
410  );
411 
412  // Perform standard decoration with a short initialization:
413  std::unique_lock<std::mutex> lk(m_lock);
414  DecorationDisposition* pTypeSubs[1 + sizeof...(Ts)] = {
415  &DecorateImmediateUnsafe(DecorationKey(auto_id<T>::key()), &immed),
416  &DecorateImmediateUnsafe(DecorationKey(auto_id<Ts>::key()), &immeds)...
417  };
418  lk.unlock();
419 
420 
421  // Pulse satisfaction:
422  MakeAtExit([this, &pTypeSubs] {
423  // Mark entries as unsatisfiable:
424  // IMPORTANT: isCheckedOut = true prevents subsequent decorations of this type
425  for(DecorationDisposition* pEntry : pTypeSubs) {
426  pEntry->m_pImmediate = nullptr;
427  pEntry->m_state = DispositionState::Unsatisfiable;
428  }
429 
430  // Now trigger a rescan to hit any deferred, unsatisfiable entries:
431 #if autowiring_USE_LIBCXX
432  for (const std::type_info* ti : {&auto_id<T>::key(), &auto_id<Ts>::key()...})
433  MarkUnsatisfiable(DecorationKey(*ti));
434 #else
435  bool dummy[] = {
436  (MarkUnsatisfiable(DecorationKey(auto_id<T>::key())), false),
437  (MarkUnsatisfiable(DecorationKey(auto_id<Ts>::key())), false)...
438  };
439  (void)dummy;
440 #endif
441  }),
442  PulseSatisfaction(pTypeSubs, 1 + sizeof...(Ts));
443  }
