261  template<class T>
262  bool Get(const T*& out, int tshift=0) const {
263  DecorationKey key(auto_id_t<T>{}, tshift);
264  const DecorationDisposition* pDisposition = GetDisposition(key);
265  if (pDisposition) {
266  switch (pDisposition->m_decorations.size()) {
267  case 0:
268  // No shared pointer decorations available, we have to try something else
269  break;
270  case 1:
271  // Single decoration, we can do what the user is asking
272  out = static_cast<const T*>(pDisposition->m_decorations[0].ptr());
273  return true;
274  default:
275  ThrowMultiplyDecoratedException(key);
276  break;
277  }
278 
279  // Second-chance satisfaction with an immediate
280  if (pDisposition->m_pImmediate) {
281  out = (T*) pDisposition->m_pImmediate;
282  return true;
283  }
284  }
285 
286  out = nullptr;
287  return false;
288  }

538  template<class T, class... Ts>
539  void DecorateImmediate(const T& immed, const Ts&... immeds) {
540  // None of the inputs may be shared pointers--if any of the inputs are shared pointers, they must be attached
541  // to this packet via Decorate, or else dereferenced and used that way.
542  static_assert(
543  !is_any<is_shared_ptr<T>::value, is_shared_ptr<Ts>::value...>::value,
544  "DecorateImmediate must not be used to attach a shared pointer, use Decorate on such a decoration instead"
545  );
546 
547  // Perform standard decoration with a short initialization:
548  std::unique_lock<std::mutex> lk(m_lock);
549  DecorationDisposition* pTypeSubs[1 + sizeof...(Ts)] = {
550  &DecorateImmediateUnsafe(DecorationKey(auto_id_t<T>(), 0), &immed),
551  &DecorateImmediateUnsafe(DecorationKey(auto_id_t<Ts>(), 0), &immeds)...
552  };
553 
554  // Pulse satisfaction:
555  MakeAtExit([this, &pTypeSubs] {
556  // Mark entries as unsatisfiable:
557  // IMPORTANT: isCheckedOut = true prevents subsequent decorations of this type
558  for(DecorationDisposition* pEntry : pTypeSubs) {
559  pEntry->m_pImmediate = nullptr;
560  pEntry->m_state = DispositionState::Complete;
561  }
562 
563  // Now trigger a rescan to hit any deferred, unsatisfiable entries:
564  autowiring::noop(
565  (MarkUnsatisfiable(DecorationKey(auto_id_t<T>(), 0)), false),
566  (MarkUnsatisfiable(DecorationKey(auto_id_t<Ts>(), 0)), false)...

568  }),
