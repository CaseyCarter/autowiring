692  template<typename T, typename... Args>
693  std::shared_ptr<T> Inject(Args&&... args) {
694  // Creator proxy, knows how to create the type we intend to inject
695  typedef autowiring::CreationRules<T, Args...> CreationRules;
696 
697  // Add this type to the TypeRegistry, also ensure that we initialize support for blind
698  // fast pointer cast to CoreObject.
699  (void) RegType<T>::r;
700  (void) autowiring::fast_pointer_cast_initializer<CoreObject, T>::sc_init;
701 
702  // First see if the base object type has already been injected. This is also necessary to
703  // ensure that a memo slot is created for the type by itself, in cases where the injected
704  // member does not inherit CoreObject and this member is eventually satisfied by one that does.
705  {
706  std::shared_ptr<T> pure;
707  FindByType(pure, true); //skip non-local slots.
708  if (pure)
709  return pure;
710  }
711 
712  // We must make ourselves current for the remainder of this call:
713  CurrentContextPusher pshr(shared_from_this());
714  std::shared_ptr<typename CreationRules::TActual> retVal(
715  CreationRules::New(*this, std::forward<Args>(args)...)
716  );
717 
718  // AutoInit if sensible to do so:
719  CallAutoInit(*retVal, has_autoinit<T>());
720 
721  try {
722  // Pass control to the insertion routine, which will handle injection from this point:
723  AddInternal(CoreObjectDescriptor(retVal, (T*)nullptr));
724  }
725  catch(autowiring_error&) {
726  // We know why this exception occurred. It's because, while we were constructing our
727  // type, someone else was constructing the same type at the same time. As a consequence,
728  // we will simply eat this exception, and handle it silently by returning the type that
729  // someone else has already attempted to construct, as per the documented behavior of
730  // Construct.
731  FindByType(retVal);
732  }
733 
734  // Factory registration if sensible to do so, but only after the underlying type has been added
735  // This ensures that any creation operations that happen as a consequence of factory registration
736  // can correctly back-reference the factory proper via autowiring
737  RegisterFactory(*retVal, autowiring::member_new_type<typename CreationRules::TActual>());
738  return std::static_pointer_cast<T>(retVal);
739  }
