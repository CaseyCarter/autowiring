645  template<typename T, typename... Args>
646  std::shared_ptr<T> Inject(Args&&... args) {
647  // Creator proxy, knows how to create the type we intend to inject
648  typedef autowiring::CreationRules<T, Args...> CreationRules;
649 
650  // Add this type to the TypeRegistry, also ensure that we initialize support for blind
651  // fast pointer cast to CoreObject.
652  (void) RegType<T>::r;
653  (void) autowiring::fast_pointer_cast_initializer<CoreObject, T>::sc_init;
654 
655  // First see if the base object type has already been injected. This is also necessary to
656  // ensure that a memo slot is created for the type by itself, in cases where the injected
657  // member does not inherit CoreObject and this member is eventually satisfied by one that does.
658  {
659  const auto& memo = FindByType(auto_id_t<T>{}, true); // Do not recurse
660  if (memo.m_value && memo.m_local)
661  return memo.m_value.template as<T>();
662  }
663 
664  // We must make ourselves current for the remainder of this call:
665  CurrentContextPusher pshr(shared_from_this());
666  std::shared_ptr<typename CreationRules::TActual> retVal(
667  CreationRules::New(*this, std::forward<Args>(args)...)
668  );
669 
670  // AutoInit if sensible to do so:
671  CallAutoInit(*retVal, has_autoinit<T>());
672 
673  try {
674  // Pass control to the insertion routine, which will handle injection from this point:
675  AddInternal(CoreObjectDescriptor(retVal, (T*)nullptr));
676  }
677  catch(autowiring_error&) {
678  // We know why this exception occurred. It's because, while we were constructing our
679  // type, someone else was constructing the same type at the same time. As a consequence,
680  // we will simply eat this exception, and handle it silently by returning the type that
681  // someone else has already attempted to construct, as per the documented behavior of
682  // Construct.
683  auto& memo = FindByType(auto_id_t<typename CreationRules::TActual>{}, true);
684  retVal = memo.m_value.template as<typename CreationRules::TActual>();
685  }
686 
687  // Factory registration if sensible to do so, but only after the underlying type has been added
688  // This ensures that any creation operations that happen as a consequence of factory registration
689  // can correctly back-reference the factory proper via autowiring
690  RegisterFactory(*retVal, autowiring::member_new_type<typename CreationRules::TActual>());
691  return std::static_pointer_cast<T>(retVal);
692  }
