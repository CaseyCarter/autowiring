682  template<typename T, typename... Args>
683  std::shared_ptr<T> Inject(Args&&... args) {
684  // Creator proxy, knows how to create the type we intend to inject
685  typedef autowiring::CreationRules<T, Args...> CreationRules;
686 
687  // Add this type to the TypeRegistry, also ensure that we initialize support for blind
688  // fast pointer cast to CoreObject.
689  (void) RegType<T>::r;
690  (void) autowiring::fast_pointer_cast_initializer<CoreObject, T>::sc_init;
691 
692  // First see if the base object type has already been injected. This is also necessary to
693  // ensure that a memo slot is created for the type by itself, in cases where the injected
694  // member does not inherit CoreObject and this member is eventually satisfied by one that does.
695  {
696  std::shared_ptr<T> pure;
697  FindByType(pure, true); //skip non-local slots.
698  if (pure)
699  return pure;
700  }
701 
702  // We must make ourselves current for the remainder of this call:
703  CurrentContextPusher pshr(shared_from_this());
704  std::shared_ptr<typename CreationRules::TActual> retVal(
705  CreationRules::New(*this, std::forward<Args>(args)...)
706  );
707 
708  // AutoInit if sensible to do so:
709  CallAutoInit(*retVal, has_autoinit<T>());
710 
711  try {
712  // Pass control to the insertion routine, which will handle injection from this point:
713  AddInternal(CoreObjectDescriptor(retVal, (T*)nullptr));
714  }
715  catch(autowiring_error&) {
716  // We know why this exception occurred. It's because, while we were constructing our
717  // type, someone else was constructing the same type at the same time. As a consequence,
718  // we will simply eat this exception, and handle it silently by returning the type that
719  // someone else has already attempted to construct, as per the documented behavior of
720  // Construct.
721  FindByType(retVal);
722  }
723 
724  // Factory registration if sensible to do so, but only after the underlying type has been added
725  // This ensures that any creation operations that happen as a consequence of factory registration
726  // can correctly back-reference the factory proper via autowiring
727  RegisterFactory(*retVal, autowiring::member_new_type<typename CreationRules::TActual>());
728  return std::static_pointer_cast<T>(retVal);
729  }
