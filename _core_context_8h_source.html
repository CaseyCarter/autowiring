635  template<typename T, typename... Args>
636  std::shared_ptr<T> Inject(Args&&... args) {
637  // Creator proxy, knows how to create the type we intend to inject
638  typedef autowiring::CreationRules<T, Args...> CreationRules;
639 
640  // Add this type to the TypeRegistry, also ensure that we initialize support for blind
641  // fast pointer cast to CoreObject.
642  (void) RegType<T>::r;
643  (void) autowiring::fast_pointer_cast_initializer<CoreObject, T>::sc_init;
644 
645  // First see if the base object type has already been injected. This is also necessary to
646  // ensure that a memo slot is created for the type by itself, in cases where the injected
647  // member does not inherit CoreObject and this member is eventually satisfied by one that does.
648  {
649  std::shared_ptr<T> pure;
650  FindByType(pure);
651  if (pure)
652  return pure;
653  }
654 
655  // We must make ourselves current for the remainder of this call:
656  CurrentContextPusher pshr(shared_from_this());
657  std::shared_ptr<typename CreationRules::TActual> retVal(
658  CreationRules::New(*this, std::forward<Args>(args)...)
659  );
660 
661  // AutoInit if sensible to do so:
662  CallAutoInit(*retVal, has_autoinit<T>());
663 
664  try {
665  // Pass control to the insertion routine, which will handle injection from this point:
666  AddInternal(ObjectTraits(retVal, (T*)nullptr));
667  }
668  catch(autowiring_error&) {
669  // We know why this exception occurred. It's because, while we were constructing our
670  // type, someone else was constructing the same type at the same time. As a consequence,
671  // we will simply eat this exception, and handle it silently by returning the type that
672  // someone else has already attempted to construct, as per the documented behavior of
673  // Construct.
674  FindByType(retVal);
675  }
676 
677  // Factory registration if sensible to do so, but only after the underlying type has been added
678  // This ensures that any creation operations that happen as a consequence of factory registration
679  // can correctly back-reference the factory proper via autowiring
680  RegisterFactory(*retVal, autowiring::member_new_type<typename CreationRules::TActual>());
681  return std::static_pointer_cast<T>(retVal);
682  }
