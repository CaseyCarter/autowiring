579  template<typename T, typename... Args>
580  std::shared_ptr<T> Inject(Args&&... args) {
581  // Creator proxy, knows how to create the type we intend to inject
582  typedef autowiring::CreationRules<T, Args...> CreationRules;
583 
584  // Add this type to the TypeRegistry, also ensure that we initialize support for blind
585  // fast pointer cast to CoreObject.
586  (void) autowiring::RegType<T>::r;
587  (void) autowiring::fast_pointer_cast_initializer<CoreObject, T>::sc_init;
588 
589  // First see if the base object type has already been injected. This is also necessary to
590  // ensure that a memo slot is created for the type by itself, in cases where the injected
591  // member does not inherit CoreObject and this member is eventually satisfied by one that does.
592  {
593  const auto& memo = FindByType(auto_id_t<T>{}, true); // Do not recurse
594  if (memo.m_value && memo.m_local)
595  return memo.m_value.template as<T>();
596  }
597 
598  // Need the memo for the actual type at this point, if we don't hold this down then this
599  // entry might not get constructed.
600  auto& memo = FindByType(auto_id_t<typename CreationRules::TActual>{}, true);
601 
602  // We must make ourselves current for the remainder of this call:
603  CurrentContextPusher pshr(shared_from_this());
604  std::shared_ptr<typename CreationRules::TActual> retVal(
605  CreationRules::New(*this, std::forward<Args>(args)...)
606  );
607  autowiring::CoreObjectDescriptor objDesc(retVal, (T*)nullptr);
608 
609  try {
610  // Pass control to the insertion routine, which will handle injection from this point:
611  AddInternal(objDesc);
612 
613  // Configure if the object is configurable, use a static check rather than checking the
614  // value of pConfigDesc because it's a little faster and one necessarily follows the other
615  // We also want this to happen before the AutoInit call is made, but after we have
616  // determined that the AutoInit call is inevitable.
617  if (autowiring::has_getconfigdescriptor<T>::value)
618  Config.Register(
619  std::static_pointer_cast<T>(retVal),
620  *objDesc.pConfigDesc
621  );
622  }
623  catch(autowiring_error&) {
624  // We know why this exception occurred. It's because, while we were constructing our
625  // type, someone else was constructing the same type at the same time. As a consequence,
626  // we will simply eat this exception, and handle it silently by returning the type that
627  // someone else has already attempted to construct, as per the documented behavior of
628  // Construct.
629  retVal = memo.m_value.template as<typename CreationRules::TActual>();
630  }
631 
632  try {
633  // AutoInit if sensible to do so, we've proven to ourselves that we are the only owner
634  // at this point
635  CallAutoInit(*retVal, autowiring::has_autoinit<T>());
636  }
637  catch (...) {
638  // The object is already in the context, it can't refuse to initialize. Throwing from
639  // here results in teardown unconditionally.
640  SignalShutdown();
641  throw;
642  }
643 
644  // Factory registration if sensible to do so, but only after the underlying type has been added
645  // This ensures that any creation operations that happen as a consequence of factory registration
646  // can correctly back-reference the factory proper via autowiring
647  RegisterFactory(*retVal, autowiring::member_new_type<typename CreationRules::TActual>());
648  return std::static_pointer_cast<T>(retVal);
649  }
