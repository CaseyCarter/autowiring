655  template<typename T, typename... Args>
656  std::shared_ptr<T> Inject(Args&&... args) {
657  // Creator proxy, knows how to create the type we intend to inject
658  typedef autowiring::CreationRules<T, Args...> CreationRules;
659 
660  // Add this type to the TypeRegistry, also ensure that we initialize support for blind
661  // fast pointer cast to CoreObject.
662  (void) RegType<T>::r;
663  (void) autowiring::fast_pointer_cast_initializer<CoreObject, T>::sc_init;
664 
665  // First see if the base object type has already been injected. This is also necessary to
666  // ensure that a memo slot is created for the type by itself, in cases where the injected
667  // member does not inherit CoreObject and this member is eventually satisfied by one that does.
668  {
669  std::shared_ptr<T> pure;
670  FindByType(pure, true); //skip non-local slots.
671  if (pure)
672  return pure;
673  }
674 
675  // We must make ourselves current for the remainder of this call:
676  CurrentContextPusher pshr(shared_from_this());
677  std::shared_ptr<typename CreationRules::TActual> retVal(
678  CreationRules::New(*this, std::forward<Args>(args)...)
679  );
680 
681  // AutoInit if sensible to do so:
682  CallAutoInit(*retVal, has_autoinit<T>());
683 
684  try {
685  // Pass control to the insertion routine, which will handle injection from this point:
686  AddInternal(CoreObjectDescriptor(retVal, (T*)nullptr));
687  }
688  catch(autowiring_error&) {
689  // We know why this exception occurred. It's because, while we were constructing our
690  // type, someone else was constructing the same type at the same time. As a consequence,
691  // we will simply eat this exception, and handle it silently by returning the type that
692  // someone else has already attempted to construct, as per the documented behavior of
693  // Construct.
694  FindByType(retVal);
695  }
696 
697  // Factory registration if sensible to do so, but only after the underlying type has been added
698  // This ensures that any creation operations that happen as a consequence of factory registration
699  // can correctly back-reference the factory proper via autowiring
700  RegisterFactory(*retVal, autowiring::member_new_type<typename CreationRules::TActual>());
701  return std::static_pointer_cast<T>(retVal);
702  }
