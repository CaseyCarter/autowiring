523  template<typename T, typename... Args>
524  std::shared_ptr<T> Construct(Args&&... args) {
525  // Add this type to the TypeRegistry
526  (void) RegType<T>::r;
527 
528  // If T doesn't inherit Object, then we need to compose a unifying type which does
529  typedef typename SelectTypeUnifier<T>::type TActual;
530  static_assert(std::is_base_of<Object, TActual>::value, "Constructive type does not implement Object as expected");
531  static_assert(
532  std::is_base_of<Object, T>::value || !has_static_new<T>::value,
533  "If type T provides a static new method, then the constructed type MUST directly inherit Object"
534  );
535 
536  // First see if the object has already been injected:
537  std::shared_ptr<TActual> retVal;
538  FindByType(retVal);
539  if(retVal)
540  return retVal;
541 
542  // We must make ourselves current for the remainder of this call:
543  CurrentContextPusher pshr(shared_from_this());
544 
545  // Cannot safely inject while holding the lock, so we have to unlock and then inject
546  retVal.reset(CreationRules::New<TActual>(std::forward<Args>(args)...));
547 
548  try {
549  // Pass control to the insertion routine, which will handle injection from this point:
550  AddInternal(AddInternalTraits(retVal, (T*)nullptr));
551  }
552  catch(autowiring_error&) {
553  // We know why this exception occurred. It's because, while we were constructing our
554  // type, someone else was constructing the same type at the same time. As a consequence,
555  // we will simply eat this exception, and handle it silently by returning the type that
556  // someone else has already attempted to construct, as per the documented behavior of
557  // Construct.
558  FindByType(retVal);
559  }
560  return retVal;
561  }
