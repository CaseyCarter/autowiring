649  template<typename T, typename... Args>
650  std::shared_ptr<T> Inject(Args&&... args) {
651  // Creator proxy, knows how to create the type we intend to inject
652  typedef autowiring::CreationRules<T, Args...> CreationRules;
653 
654  // Add this type to the TypeRegistry, also ensure that we initialize support for blind
655  // fast pointer cast to CoreObject.
656  (void) RegType<T>::r;
657  (void) autowiring::fast_pointer_cast_initializer<CoreObject, T>::sc_init;
658 
659  // First see if the base object type has already been injected. This is also necessary to
660  // ensure that a memo slot is created for the type by itself, in cases where the injected
661  // member does not inherit CoreObject and this member is eventually satisfied by one that does.
662  {
663  std::shared_ptr<T> pure;
664  FindByType(pure, true); //skip non-local slots.
665  if (pure)
666  return pure;
667  }
668 
669  // We must make ourselves current for the remainder of this call:
670  CurrentContextPusher pshr(shared_from_this());
671  std::shared_ptr<typename CreationRules::TActual> retVal(
672  CreationRules::New(*this, std::forward<Args>(args)...)
673  );
674 
675  // AutoInit if sensible to do so:
676  CallAutoInit(*retVal, has_autoinit<T>());
677 
678  try {
679  // Pass control to the insertion routine, which will handle injection from this point:
680  AddInternal(CoreObjectDescriptor(retVal, (T*)nullptr));
681  }
682  catch(autowiring_error&) {
683  // We know why this exception occurred. It's because, while we were constructing our
684  // type, someone else was constructing the same type at the same time. As a consequence,
685  // we will simply eat this exception, and handle it silently by returning the type that
686  // someone else has already attempted to construct, as per the documented behavior of
687  // Construct.
688  FindByType(retVal);
689  }
690 
691  // Factory registration if sensible to do so, but only after the underlying type has been added
692  // This ensures that any creation operations that happen as a consequence of factory registration
693  // can correctly back-reference the factory proper via autowiring
694  RegisterFactory(*retVal, autowiring::member_new_type<typename CreationRules::TActual>());
695  return std::static_pointer_cast<T>(retVal);
696  }
