510  template<typename T, typename... Args>
511  std::shared_ptr<T> Inject(Args&&... args) {
512  // Creator proxy, knows how to create the type we intend to inject
513  typedef autowiring::CreationRules<T, Args...> CreationRules;
514 
515  // Add this type to the TypeRegistry, also ensure that we initialize support for blind
516  // fast pointer cast to Object.
517  (void) RegType<T>::r;
518  (void) autowiring::fast_pointer_cast_initializer<Object, T>::sc_init;
519 
520  // First see if the base object type has already been injected. This is also necessary to
521  // ensure that a memo slot is created for the type by itself, in cases where the injected
522  // member does not inherit Object and this member is eventually satisfied by one that does.
523  {
524  std::shared_ptr<T> pure;
525  FindByType(pure);
526  if (pure)
527  return pure;
528  }
529 
530  // We must make ourselves current for the remainder of this call:
531  CurrentContextPusher pshr(shared_from_this());
532  std::shared_ptr<typename CreationRules::TActual> retVal(
533  CreationRules::New(*this, std::forward<Args>(args)...)
534  );
535 
536  // AutoInit if sensible to do so:
537  CallAutoInit(*retVal, has_autoinit<T>());
538 
539  try {
540  // Pass control to the insertion routine, which will handle injection from this point:
541  AddInternal(ObjectTraits(retVal, (T*)nullptr));
542  }
543  catch(autowiring_error&) {
544  // We know why this exception occurred. It's because, while we were constructing our
545  // type, someone else was constructing the same type at the same time. As a consequence,
546  // we will simply eat this exception, and handle it silently by returning the type that
547  // someone else has already attempted to construct, as per the documented behavior of
548  // Construct.
549  FindByType(retVal);
550  }
551 
552  // Factory registration if sensible to do so, but only after the underlying type has been added
553  // This ensures that any creation operations that happen as a consequence of factory registration
554  // can correctly back-reference the factory proper via autowiring
555  RegisterFactory(*retVal, autowiring::member_new_type<typename CreationRules::TActual>());
556  return std::static_pointer_cast<T>(retVal);
557  }
