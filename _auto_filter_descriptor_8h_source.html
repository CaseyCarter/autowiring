147 template<class T, class... Args>
148 struct CallExtractor<Deferred (T::*)(Args...)>:
149  Decompose<void (T::*)(Args...)>
150 {
151  static const bool deferred = true;
152  static const size_t N = sizeof...(Args);
153 
154  template<Deferred(T::*memFn)(Args...)>
155  static void Call(void* pObj, AutoPacket& autoPacket) {
156  // Obtain a shared pointer of the AutoPacket in order to ensure the packet
157  // stays resident when we pend this lambda to the destination object's
158  // dispatch queue.
159  auto pAutoPacket = autoPacket.shared_from_this();
160 
161  // Pend the call to this object's dispatch queue:
162  *(T*) pObj += [pObj, pAutoPacket] {
163  (((T*) pObj)->*memFn)(
164  subscriber_traits<Args>()(*pAutoPacket)...
165  );
166  };
