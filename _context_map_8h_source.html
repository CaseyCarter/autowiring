93  context->AddTeardownListener([this, key, tracker] {
94  // Prevent the map from being deleted while we process this teardown notice:
95  auto locked = tracker.lock();
96  if(!locked)
97  // Context survived the map
98  return;
99 
100  std::lock_guard<std::mutex> lk(*locked);
101  if(locked->destroyed)
102  // Map passed through teardown pathway while we were trying to lock it
103  return;
104 
105  // We only remove the key if it's expired. Under normal circumstances, the key will
106  // be expired by the time we get here, but there is a small chance that the same key
107  // will be introduced at the exact time that the context is tearing down, which could
108  // cause the slot for that key to be reclaimed earlier than expected.
109  auto q = m_contexts.find(key);
110 
111  // There is a very unlikely race which could cause the key to not be found. It involves
112  // a second context being created and destroyed on the same key as some original context.
113  // This causes one of the two context's teardown handlers to attempt to evict the same
114  // key at the same time, and one of them will of course fail.
115  if(q == m_contexts.end())
116  return;
117 
118  // Try to lock--potentially, this key has been reclaimed by a different context, and in
119  // that case, the new context will gain the responsibility of tearing down this key when
120  // the time comes.
121  auto sp = q->second.lock();
122  if(!sp)
123  this->m_contexts.erase(key);
124  });
125  }
